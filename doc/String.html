def color_chooser(p)
    case p
    when "B"
        blue
    when "K"
        black
    when "C"
        cyan
    when "G"
        green
    when "M"
        magenta
    when "O"
        white
    when "R"
        red      
    when "Y"
        yellow
    end
end
